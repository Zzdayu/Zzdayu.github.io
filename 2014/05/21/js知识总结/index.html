布尔类型转化数值类型

var a = true;
var b = false;
console.log(Number(a));//NaN
console.log( Number(b));//NaN

如果布尔类型转为数值类型, true为1, false为0;

任意的javascript的值都可以转换成布尔值。下面这些值会被转换成false:
undefined,null,0,-0,NaN,”” //空字符串;

需要注意的是’0’这个仅仅包含0的字符串会转成true！

补充

null == undefined //true 
null === undefined //false

数值参与逻辑运算

数值参与&&,如果结果为真,返回第二个为真的值,如果结果为假,返回第一个为假的值;

var a = 2;
var b = 3;
console.log(a && b)//3
console.log(0 && a)//0

数值参与||，如果结果为真，返回第一个为真的值，如果为假返回第二个为假的值

如果有变量参与&&运算符，会把两个表达式都计算

如果参与||,那么第一个为真值，第二个表达式不会计算，如果第一个为假，会计算第二个的值

数组的常用的方法
http://www.w3school.com.cn/jsref/jsref_obj_array.asp

返回值 arr.push() | arr.unshift()

//1.创建数组
var numArray = new Array();

//2.添加元素
var a = numArray.push(1);
var b =  numArray.push(2);

console.log(numArray);
console.log(a);//1
console.log(b);//2

//push会在数组的结尾处添加一个元素，改变原来的数组,返回值是数组的长度

//2.1第二种添加元素
var aa = numArray.unshift(0);
console.log(numArray);
console.log(aa);

//unshift：会在数组的开始添加一个元素，会改变数组，返回值是数组的长度

数组删除方法 arr.pop | arr.shift：

var numArray = [1, 2, 3];

//1.删除元素
var a = numArray.pop();
console.log(a);    //返回被删除的元素3
console.log(numArray);    //[1,2]
//从数组的最后删除元素，返回删除的元素

//从开头删除元素
var aa = numArray.shift();
console.log(aa);    //1
console.log(numArray);    //[2]
//shift会改变数组，返回被删除的元素

链接数组： arr1.concat(arr2)

/*链接数组方法 concat*/
var array1 = [1,2,3];
var array2 = [5,6,7];

var allArray = array1.concat(array2);

console.log(allArray);//[1, 2, 3, 5, 6, 7]
console.log(array1);//[1, 2, 3]
console.log(array2);//[5, 6, 7]

数组转化: arr.join() | arr.split()

var array = [1, 3, 5];

//数组转化字符
var arrStr = array.join('-');
console.log(arrStr);//1-3-5
//如果join方法中传入什么就会把数组用什么连接起来

//字符串转化数组
var newArray = arrStr.split('-');
console.log(newArray);//["1", "3", "5"]
//split会把字符转化成数组，但是这个字符必须是有规律的

数组排序 arr.sort(fn) | arr.reverse()

使用方法:
var numArray  = [1, 3, 5, 4];

numArray.sort(function (a, b){
    return a > b ? 1 : -1;
});

console.log(numArray);//[1, 3, 4, 5]
若 a 小于 b，在排序后的数组中 a 应该出现在 b 之前，则返回一个小于 0 的值。
若 a 等于 b，则返回 0。
若 a 大于 b，则返回一个大于 0 的值。

颠倒数组

console.log(numArray.reverse());//[5, 4, 3, 1]
//注释：该方法会改变原来的数组，而不会创建新的数组。

获取某一个位置的字符 charAt()

var charStr = 'my name is zz';
console.log(charStr.length);//13
//空格是占用位置
//获取元素
var sym =  charStr.charAt(1);
console.log(sym);//y

获取一个字符的位置 str.indexOf()

var charString = 'my is';
//获取某一个字符的位置
var x = charString.indexOf('i');
var y = charString.indexOf('n');
console.log(x);//3
console.log(y);//找不到返回 -1

//indexOf:会获取传入的字符的位置，位置从左到右边，如果获取到第一个元素后，就会返回第一个获取元素的位置,如果没有对应的元素就会返回-1；

var newPoint = charString.lastIndexOf('s');
console.log(newPoint);//4
//lastIndexOf：从后往前找对应的元素，如果找到就会把对应的位置返回，但是注意位置排序还是从左到右

字符大小写转化

var lowString = 'abc cddfff';

//转化成大写
var newLowString = lowString.toUpperCase();
console.log(newLowString);//ABC CDDFFF

//大小写方法不会改变原来的字符，返回值是对应转化后的结果
var newString2 = newLowString.toLowerCase();
console.log(newString2);//abc cddfff
